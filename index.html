Sub GenererRentabiliteReel()
  ' Analyse rentabilité réelle par commande
  Dim coutMO As Double
  Dim coutST As Double
  Dim coutComposants As Double

  ' Cumul des coûts sous-produits
  coutMO = GetCoutCentreCharge(produit)
  coutST = GetCoutSousTraitance(produit)
  coutComposants = GetCoutComposants(produit)

  rentabilite = prixVente - (coutMO + coutST _
    + coutComposants)
  ' Export vers Power BI →
End Sub
